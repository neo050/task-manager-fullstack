### 1 · PostgreSQL & migration (Deep-Work #1)

```bash
# create Docker compose if missing
cat > docker-compose.yml <<'YAML'
services:
  db:
    image: postgres:15
    environment:
      POSTGRES_USER: task_user
      POSTGRES_PASSWORD: task_pass
      POSTGRES_DB: task_db
    ports:
      - "5432:5432"
    volumes:
      - db_data:/var/lib/postgresql/data
volumes:
  db_data:
YAML

docker compose up -d db

# install pg library
cd server && npm i pg dotenv

# create migration script
cat > server/sql/init.sql <<'SQL'
CREATE TABLE IF NOT EXISTS users (
  id SERIAL PRIMARY KEY,
  email TEXT UNIQUE NOT NULL,
  password_hash TEXT NOT NULL,
  created_at TIMESTAMP DEFAULT NOW()
);
SQL

# run migration once (simple psql)
docker compose exec db psql -U task_user -d task_db -f /var/lib/postgresql/data/../init.sql
git add docker-compose.yml server/sql/init.sql
git commit -m "feat(db): add postgres service and users table"


```
